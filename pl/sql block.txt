suppose we have emp table
that has ENAME column


CREATE PROCEDURE P1()
BEGIN
    DECLARE emp_name VARCHAR(50);
    SELECT ENAME INTO emp_name
    FROM emp
    WHERE empno = 7369;

    select emp_name;

END
