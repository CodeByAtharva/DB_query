create procedure get_emp3()
BEGIN
declare name varchar(255);
declare emp_cursor CURSOR for select ENAME from emp;
open emp_cursor;

fetch emp_cursor into name;

select CONCAT("First Employee name:",name) as ename;

close emp_cursor;
